@dataclass
class PrerequisiteCheck:
    name: str
    status: Literal["pass", "warn", "fail"]
    message: str

@dataclass
class ProviderContext:
    metadata: Any  # WorkflowMetadata
    output_dir: Path
    stage: str | None
    workflow_path: Path
    definition: Any = None
    auto_approve: bool = False
    stage_var_file: Path | None = None

class InfrastructureProvider(ABC):
    @abstractmethod
    def generate(self, ctx: ProviderContext) -> None: ...
    @abstractmethod
    def deploy(self, ctx: ProviderContext) -> None: ...
    @abstractmethod
    def teardown(self, ctx: ProviderContext) -> None: ...
    @abstractmethod
    def check_prerequisites(self, ctx: ProviderContext) -> list[PrerequisiteCheck]: ...
    @abstractmethod
    def validate_config(self, ctx: ProviderContext) -> None: ...

    def run_provider_command(self, cmd, cwd, env) -> subprocess.CompletedProcess[str]:
        """shell=False, capture_output=True"""
```

class FileTransport(MetadataTransport):
    def prepare(self, metadata, env) -> list[str]:  # sets RSF_METADATA_FILE in env, returns []
    def cleanup(self) -> None:  # removes temp file

class EnvTransport(MetadataTransport):
    def prepare(self, metadata, env) -> list[str]:  # sets RSF_WORKFLOW_NAME, RSF_STAGE, RSF_METADATA_JSON
    def cleanup(self) -> None:  # no-op

class ArgsTransport(MetadataTransport):
    def __init__(self, arg_templates: list[str]):  # validates templates at construction
    def prepare(self, metadata, env) -> list[str]:  # returns substituted args
    def cleanup(self) -> None:  # no-op
```

def run_provider_command_streaming(self, cmd, cwd, env) -> subprocess.CompletedProcess[str]:
    """shell=False, no capture_output, streams stdout/stderr to terminal"""
    merged_env = {**os.environ, **(env or {})}

    CustomProvider class (src/rsf/providers/custom.py):
    - Inherits from InfrastructureProvider
    - Imports: os, logging, subprocess, Path from pathlib, CustomProviderConfig from rsf.dsl.models,
      InfrastructureProvider/PrerequisiteCheck/ProviderContext from rsf.providers.base,
      FileTransport/EnvTransport/ArgsTransport from rsf.providers.transports

    Methods:

    generate(ctx: ProviderContext) -> None:
      - No-op. Custom providers don't generate infrastructure code.
      - Log at debug level: "Custom provider: generate is a no-op"

    deploy(ctx: ProviderContext) -> None:
      1. config = self._get_config(ctx) -- extracts CustomProviderConfig from ProviderContext
      2. program = self._validate_program(config.program) -- validates absolute, exists, executable
      3. transport = self._create_transport(config) -- creates appropriate MetadataTransport
      4. env: dict[str, str] = {}
      5. if config.env: env.update(config.env)
      6. try:
           extra_args = transport.prepare(ctx.metadata, env)
           cmd = [str(program)] + list(config.args) + extra_args
           self.run_provider_command_streaming(cmd, cwd=ctx.workflow_path.parent, env=env)
         finally:
           transport.cleanup()

    teardown(ctx: ProviderContext) -> None:
      1. config = self._get_config(ctx)
      2. if config.teardown_args is None:
           raise NotImplementedError("Custom provider teardown not configured. Set 'teardown_args' in the custom provider config to enable teardown.")
      3. program = self._validate_program(config.program)
      4. transport = self._create_transport(config)
      5. env: dict[str, str] = {}
      6. if config.env: env.update(config.env)
      7. try:
           extra_args = transport.prepare(ctx.metadata, env)
           cmd = [str(program)] + list(config.teardown_args) + extra_args
           self.run_provider_command_streaming(cmd, cwd=ctx.workflow_path.parent, env=env)
         finally:
           transport.cleanup()

    check_prerequisites(ctx: ProviderContext) -> list[PrerequisiteCheck]:
      1. config = self._get_config(ctx)
      2. checks: list[PrerequisiteCheck] = []
      3. Check if program path exists:
         - Path(config.program).exists() -> pass: "custom program found: {program}"
         - Not exists -> fail: "custom program not found: {program}"
      4. Check if program is executable (only if exists):
         - os.access(config.program, os.X_OK) -> pass: "custom program is executable"
         - Not executable -> fail: "custom program is not executable: {program}. Fix: chmod +x {program}"
      5. Return checks

    validate_config(ctx: ProviderContext) -> None:
      1. config = self._get_config(ctx)
      2. if not Path(config.program).is_absolute():
           raise ValueError(f"Custom provider program must be an absolute path, got: {config.program}")
      3. Validate arg templates if metadata_transport is "args":
         Use ArgsTransport constructor to validate -- it raises ValueError on invalid placeholders
         ArgsTransport(config.args)  -- let the ValueError propagate if templates are invalid
      4. If teardown_args is not None and metadata_transport is "args":
         ArgsTransport(config.teardown_args)  -- validate teardown templates too

    _get_config(ctx: ProviderContext) -> CustomProviderConfig:
      - Access the custom provider config from the infrastructure config.
      - The infrastructure config is available via the workflow definition.
      - For the custom provider, the config should be passed through ProviderContext.
      - APPROACH: Read from the workflow YAML infrastructure block. The deploy_cmd already parses
        InfrastructureConfig and passes it through. However, ProviderContext does not carry the
        InfrastructureConfig directly.
      - PRACTICAL APPROACH: Since ProviderContext.metadata contains WorkflowMetadata which doesn't
        carry provider config, and the provider is instantiated by the registry with no args,
        the config must come from somewhere else.
      - SOLUTION: Accept config via constructor. CustomProvider.__init__(self, config: CustomProviderConfig | None = None).
        The registry pattern calls cls() with no args, so default to None.
        When config is None, raise ValueError("CustomProvider requires config. Pass CustomProviderConfig to constructor.").
        UPDATE: Actually looking at the deploy_cmd code, get_provider() returns a zero-arg instance.
        The config needs to come from ProviderContext somehow. The cleanest approach:
        Store config on the instance when validate_config() or deploy() first reads it from the
        parsed InfrastructureConfig.
      - REVISED SOLUTION: The infrastructure config is resolved in deploy_cmd via resolve_infra_config().
        The InfrastructureConfig is available at that point. We need a way to pass the custom config
        to the provider. Two options:
        (a) Add an optional infra_config field to ProviderContext (extends existing dataclass)
        (b) Have CustomProvider._get_config read the workflow YAML and parse it itself
        Option (a) is cleaner. Add infra_config: Any = None to ProviderContext.
        BUT: Adding to ProviderContext changes the base class, affecting all providers.
        Option (c): Use the existing definition field on ProviderContext. The StateMachineDefinition
        has an infrastructure field. Access ctx.definition.infrastructure.custom.
        This requires ctx.definition to be set, which it is in deploy_cmd (it is parsed there).
        BUT: Check deploy_cmd -- does it pass definition to ProviderContext?
        Looking at deploy_cmd: ctx = ProviderContext(metadata=..., output_dir=..., stage=..., workflow_path=...)
        It does NOT pass definition. definition defaults to None.
        BEST SOLUTION: Have deploy_cmd pass the definition to ProviderContext.definition.
        Then _get_config reads ctx.definition.infrastructure.custom.
        However, this requires a change in deploy_cmd.py (Plan 03 territory).
        FOR THIS PLAN: implement _get_config to read from ctx.definition.infrastructure.custom
        and raise a clear error if ctx.definition is None or ctx.definition.infrastructure is None
        or ctx.definition.infrastructure.custom is None.

      Implementation:
      - if ctx.definition is None or not hasattr(ctx.definition, 'infrastructure'):
          raise ValueError("CustomProvider requires workflow definition in ProviderContext")
      - if ctx.definition.infrastructure is None or ctx.definition.infrastructure.custom is None:
          raise ValueError("CustomProvider requires 'custom' block in infrastructure config")
      - return ctx.definition.infrastructure.custom

    _validate_program(program: str) -> Path:
      - path = Path(program)
      - if not path.is_absolute(): raise ValueError(f"Custom provider program must be an absolute path, got: {program}")
      - if not path.exists(): raise FileNotFoundError(f"Custom provider program not found: {program}")
      - if not os.access(path, os.X_OK): raise PermissionError(f"Custom provider program is not executable: {program}\nFix: chmod +x {program}")
      - return path

    _create_transport(config: CustomProviderConfig) -> MetadataTransport:
      - if config.metadata_transport == "file": return FileTransport()
      - if config.metadata_transport == "env": return EnvTransport()
      - if config.metadata_transport == "args": return ArgsTransport(config.args)
      - raise ValueError(f"Unknown metadata_transport: {config.metadata_transport}")

    run_provider_command_streaming(cmd, cwd, env):
      - Copy the pattern from CDKProvider.run_provider_command_streaming exactly:
        logger.info("Running (streaming): %s", " ".join(cmd))
        merged_env = {**os.environ, **(env or {})}
        return subprocess.run(cmd, cwd=cwd, env=merged_env, check=True, text=True, shell=False)
