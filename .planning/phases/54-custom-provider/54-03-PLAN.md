    test_custom_deploy_file_transport_end_to_end:
      1. Create a real temp script (#!/bin/sh\necho "deployed"\nexit 0) with chmod +x
      2. Create InfrastructureConfig with provider="custom" and CustomProviderConfig with
         program=temp_script, args=["deploy"], metadata_transport="file"
      3. Create a minimal StateMachineDefinition with infrastructure set

      Same as above but metadata_transport="env"
      Assert: RSF_WORKFLOW_NAME, RSF_STAGE, RSF_METADATA_JSON in env

    test_custom_deploy_validates_executable:

      Assert: deploy() raises PermissionError
