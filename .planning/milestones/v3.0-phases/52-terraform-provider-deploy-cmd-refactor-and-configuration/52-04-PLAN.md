    - Error messages use infrastructure.field_path format (e.g., "infrastructure.provider: Unknown provider 'pulumi'")

    ## Step 1: Update validate_cmd.py

    After the existing step 4 (semantic validation), add step 5 (infrastructure validation):

    ```python
    # 5. Infrastructure config validation
    # Check workflow YAML infrastructure block (already validated by Pydantic in step 3)
    # Now check provider name is registered
    if definition.infrastructure is not None:
        try:
            from rsf.providers import get_provider
            get_provider(definition.infrastructure.provider)
        except ProviderNotFoundError:
            from rsf.providers import list_providers
            available = list_providers()
            console.print(f"[red]Validation errors in[/red] {workflow}:")
            console.print(
                f"  [yellow]infrastructure.provider[/yellow]: "
                f"Unknown provider '{definition.infrastructure.provider}'. "
                f"Available: {available}"
            )
            raise typer.Exit(code=1)

    # 6. Check rsf.toml if exists (validate even if workflow YAML has no infrastructure block)
    from rsf.config import load_project_config
    try:
        toml_config = load_project_config(workflow.parent)
        if toml_config is not None:
            try:
                from rsf.providers import get_provider
                get_provider(toml_config.provider)
            except ProviderNotFoundError:
                from rsf.providers import list_providers
                available = list_providers()
                console.print(f"[red]Validation errors in[/red] rsf.toml:")
                console.print(
                    f"  [yellow]infrastructure.provider[/yellow]: "
                    f"Unknown provider '{toml_config.provider}'. "
                    f"Available: {available}"
                )
                raise typer.Exit(code=1)
    except (ValueError, ValidationError) as exc:
        console.print(f"[red]Validation errors in[/red] rsf.toml:")
        console.print(f"  {exc}")
        raise typer.Exit(code=1)
    ```
