    The generated code structure when tracing=True should be:
    ```python
    def lambda_handler(context: DurableContext, event: dict) -> dict:
        # ... startup hooks ...
        current_state = "StartState"
        input_data = event

        _span_ctx = None
        if _OTEL_AVAILABLE:
            _span_ctx = _tracer.start_span(
                "workflow.execute",
                attributes={"workflow.name": "...", "workflow.start_at": "..."}
            )
            _span_ctx.__enter__()

        try:
            while current_state is not None:
                _state_span = None
                if _OTEL_AVAILABLE:
                    _state_span = _tracer.start_span(current_state)
                    _state_span.__enter__()
                try:
                    # ... state dispatch (if/elif blocks) ...
                finally:
                    if _state_span is not None:
                        _state_span.__exit__(None, None, None)
        except Exception as _wf_err:
            if _OTEL_AVAILABLE and _span_ctx is not None:
                _span_ctx.set_status(StatusCode.ERROR, str(_wf_err))
            raise
        finally:
            if _span_ctx is not None:
                _span_ctx.__exit__(None, None, None)

        return input_data
    ```

    IMPORTANT: Use explicit __enter__/__exit__ rather than context managers (with statements) because the existing code structure uses a while loop with if/elif chains — nesting context managers inside that would change indentation of all existing state blocks. The explicit approach avoids restructuring.
