    class ErrorRateAlarm(BaseModel):
        """CloudWatch error rate alarm configuration."""
        model_config = {"extra": "forbid"}
        type: Literal["error_rate"]
        threshold: float = Field(gt=0)  # Error percentage threshold (e.g., 5 means 5%)
        period: int = Field(default=300, ge=60)  # Evaluation period in seconds
        evaluation_periods: int = Field(default=1, ge=1)
        sns_topic_arn: str | None = None  # Optional: use existing SNS topic

    class DurationAlarm(BaseModel):
        """CloudWatch duration alarm configuration."""
        model_config = {"extra": "forbid"}
        type: Literal["duration"]
        threshold: float = Field(gt=0)  # Duration threshold in milliseconds
        period: int = Field(default=300, ge=60)
        evaluation_periods: int = Field(default=1, ge=1)
        sns_topic_arn: str | None = None

    class ThrottleAlarm(BaseModel):
        """CloudWatch throttle alarm configuration."""
        model_config = {"extra": "forbid"}
        type: Literal["throttle"]
        threshold: float = Field(gt=0)  # Throttle count threshold
        period: int = Field(default=300, ge=60)
        evaluation_periods: int = Field(default=1, ge=1)
        sns_topic_arn: str | None = None
    ```

    5. Create src/rsf/terraform/templates/alarms.tf.j2 using HCL Jinja2 delimiters (<< >>, <% %>, <# #>):

    The template logic:
    - If ANY alarm has no sns_topic_arn, create a shared SNS topic for RSF-generated notifications
    - Each alarm type maps to a CloudWatch metric:
      - error_rate -> Errors metric with Statistic "Sum" and ComparisonOperator "GreaterThanOrEqualToThreshold"
      - duration -> Duration metric with Statistic "Average" and ComparisonOperator "GreaterThanOrEqualToThreshold"
      - throttle -> Throttles metric with Statistic "Sum" and ComparisonOperator "GreaterThanOrEqualToThreshold"
    - Namespace: "AWS/Lambda", Dimension: FunctionName = local.function_name

    ```

    <% set needs_sns_topic = alarms | selectattr('sns_topic_arn', 'none') | list | length > 0 %>
    <% if needs_sns_topic %>
    resource "aws_sns_topic" "<< resource_id >>_alarm_notifications" {
      name = "${local.function_name}-alarm-notifications"
    }
    <% endif %>

    <% for alarm in alarms %>
    <% if alarm.type == "error_rate" %>
    resource "aws_cloudwatch_metric_alarm" "<< resource_id >>_error_rate_<< loop.index0 >>" {
      alarm_name          = "${local.function_name}-error-rate-<< loop.index0 >>"
      comparison_operator = "GreaterThanOrEqualToThreshold"
      evaluation_periods  = << alarm.evaluation_periods >>
      metric_name         = "Errors"
      namespace           = "AWS/Lambda"
      period              = << alarm.period >>
      statistic           = "Sum"
      threshold           = << alarm.threshold >>
      alarm_description   = "Error rate alarm for ${local.function_name}"
      treat_missing_data  = "notBreaching"

        FunctionName = aws_lambda_function.<< resource_id >>.function_name
      }

    <% if alarm.sns_topic_arn %>
