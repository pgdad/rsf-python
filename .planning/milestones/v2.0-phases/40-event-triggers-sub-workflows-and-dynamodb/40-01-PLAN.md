    1. In src/rsf/dsl/models.py, add three new Pydantic models following the LambdaUrlConfig pattern:

    ```python
    class EventBridgeTrigger(BaseModel):
        """EventBridge rule trigger configuration."""
        model_config = {"extra": "forbid"}
        type: Literal["eventbridge"]
        schedule_expression: str | None = None  # e.g., "rate(5 minutes)", "cron(0 12 * * ? *)"
        event_pattern: dict[str, Any] | None = None  # EventBridge event pattern JSON

    class SQSTrigger(BaseModel):
        """SQS queue trigger configuration."""
        model_config = {"extra": "forbid"}
        type: Literal["sqs"]
        queue_name: str
        batch_size: int = Field(default=10, ge=1, le=10000)

    class SNSTrigger(BaseModel):
        """SNS subscription trigger configuration."""
        model_config = {"extra": "forbid"}
        type: Literal["sns"]
        topic_arn: str
    ```

    2. Create a discriminated union type for triggers:
    ```python
    TriggerConfig = Annotated[
        Union[EventBridgeTrigger, SQSTrigger, SNSTrigger],
        Field(discriminator="type"),
    ]
    ```

    Place the union in models.py (or in __init__.py following the State pattern — use whichever is simpler since trigger types don't have circular dependencies).

    ```
    # DO NOT EDIT - Generated by RSF

    <% for trigger in triggers %>
    <% if trigger.type == "eventbridge" %>
    resource "aws_cloudwatch_event_rule" "<< resource_id >>_trigger_<< loop.index0 >>" {
      name                = "${local.function_name}-trigger-<< loop.index0 >>"
      <% if trigger.schedule_expression %>
      schedule_expression = "<< trigger.schedule_expression >>"
      <% endif %>
      <% if trigger.event_pattern %>
      event_pattern = jsonencode(<< trigger.event_pattern | tojson >>)
      <% endif %>
    }

    resource "aws_cloudwatch_event_target" "<< resource_id >>_target_<< loop.index0 >>" {
      rule = aws_cloudwatch_event_rule.<< resource_id >>_trigger_<< loop.index0 >>.name
      arn  = aws_lambda_function.<< resource_id >>.arn
    }

    resource "aws_lambda_permission" "<< resource_id >>_allow_eventbridge_<< loop.index0 >>" {
      statement_id  = "AllowEventBridgeInvoke<< loop.index0 >>"
      action        = "lambda:InvokeFunction"
      function_name = aws_lambda_function.<< resource_id >>.function_name
      principal     = "events.amazonaws.com"
      source_arn    = aws_cloudwatch_event_rule.<< resource_id >>_trigger_<< loop.index0 >>.arn
    }
    <% elif trigger.type == "sqs" %>
    resource "aws_sqs_queue" "<< resource_id >>_trigger_<< loop.index0 >>" {
      name = "<< trigger.queue_name >>"
    }

    resource "aws_lambda_event_source_mapping" "<< resource_id >>_sqs_<< loop.index0 >>" {
      event_source_arn = aws_sqs_queue.<< resource_id >>_trigger_<< loop.index0 >>.arn
      function_name    = aws_lambda_function.<< resource_id >>.arn
      batch_size       = << trigger.batch_size >>
    }
    <% elif trigger.type == "sns" %>
    resource "aws_sns_topic_subscription" "<< resource_id >>_trigger_<< loop.index0 >>" {
      topic_arn = "<< trigger.topic_arn >>"
      protocol  = "lambda"
      endpoint  = aws_lambda_function.<< resource_id >>.arn
    }

    resource "aws_lambda_permission" "<< resource_id >>_allow_sns_<< loop.index0 >>" {
      statement_id  = "AllowSNSInvoke<< loop.index0 >>"
      action        = "lambda:InvokeFunction"
      function_name = aws_lambda_function.<< resource_id >>.function_name
      principal     = "sns.amazonaws.com"
      source_arn    = "<< trigger.topic_arn >>"
    }
    <% endif %>
    <% endfor %>
    ```
